 90 	/**
 91 	 * Calls the super class refresh method
 92 	 *  and updates the svg elements that are referenced by a property.
 93 	 */
 94 	refresh: function() {
 95 		//call base class refresh method
 96 		arguments.callee.$.refresh.apply(this, arguments);
 97 		
 98 		if(this.node.ownerDocument) {
 99 			//adjust SVG to properties' values
100 			var me = this;
101 			this.propertiesChanged.each((function(propChanged) {
102 				if(propChanged.value) {
103 					var prop = this.properties[propChanged.key];
104 					var property = this.getStencil().property(propChanged.key);
105 					this.propertiesChanged[propChanged.key] = false;
106 
107 					//handle choice properties
108 					if(property.type() == ORYX.CONFIG.TYPE_CHOICE) {
109 						//iterate all references to SVG elements
110 						property.refToView().each((function(ref) {
111 							//if property is referencing a label, update the label
112 							if(ref !== "") {
113 								var label = this._labels[this.id + ref];
114 								if (label && property.item(prop)) {
115 									label.text(property.item(prop).title());
116 								}
117 							}
118 						}).bind(this));
119 							
120 						//if the choice's items are referencing SVG elements
121 						// show the selected and hide all other referenced SVG
122 						// elements
123 						var refreshedSvgElements = new Hash();
124 						property.items().each((function(item) {
125 							item.refToView().each((function(itemRef) {
126 								if(itemRef == "") { return; }
127 								
128 								var svgElem = this.node.ownerDocument.getElementById(this.id + itemRef);
129 	
130 								if(!svgElem) { return; }
131 								
132 								
133 								/* Do not refresh the same svg element multiple times */
134 								if(!refreshedSvgElements[svgElem.id] || prop == item.value()) {
135 									svgElem.setAttributeNS(null, 'display', ((prop == item.value()) ? 'inherit' : 'none'));
136 									refreshedSvgElements[svgElem.id] = svgElem;
137 								}
138 								
139 								// Reload the href if there is an image-tag
140 								if(ORYX.Editor.checkClassType(svgElem, SVGImageElement)) {
141 									svgElem.setAttributeNS('http://www.w3.org/1999/xlink', 'href', svgElem.getAttributeNS('http://www.w3.org/1999/xlink', 'href'));
142 								}
143 							}).bind(this));
144 						}).bind(this));
145 						
146 					} else { //handle properties that are not of type choice
147 						//iterate all references to SVG elements
148 						property.refToView().each((function(ref) {
149 							//if the property does not reference an SVG element,
150 							// do nothing
151 
152 							if(ref === "") { return; }
153 		
154 							var refId = this.id + ref;
155 
156 							//get the SVG element
157 							var svgElem = this.node.ownerDocument.getElementById(refId);
158 
159 							//if the SVG element can not be found
160 							if(!svgElem || !(svgElem.ownerSVGElement)) { 
161 								//if the referenced SVG element is a SVGAElement, it cannot
162 								// be found with getElementById (Firefox bug).
163 								// this is a work around
164 								if(property.type() === ORYX.CONFIG.TYPE_URL || property.type() === ORYX.CONFIG.TYPE_DIAGRAM_LINK) {
165 									var svgElems = this.node.ownerDocument.getElementsByTagNameNS('http://www.w3.org/2000/svg', 'a');
166 									
167 									svgElem = $A(svgElems).find(function(elem) {
168 										return elem.getAttributeNS(null, 'id') === refId;
169 									});
170 									
171 									if(!svgElem) { return; } 
172 								} else {
173 									//this.propertiesChanged[propChanged.key] = true;
174 									return;
175 								}					
176 							}
177 							
178 							if (property.complexAttributeToView()) {
179 								var label = this._labels[refId];
180 								if (label) {
181 									try {
182 								    	propJson = prop.evalJSON();
183 								    	var value = propJson[property.complexAttributeToView()]
184 								    	label.text(value ? value : prop);
185 								    } catch (e) {
186 								    	label.text(prop);
187 								    }
188 								}
189 								
190 							} else {
191 
192 								switch (property.type()) {
193 									case ORYX.CONFIG.TYPE_BOOLEAN:	
194 										
195 										if (typeof prop == "string")
196 											prop = prop === "true"
197 	
198 										svgElem.setAttributeNS(null, 'display', (!(prop === property.inverseBoolean())) ? 'inherit' : 'none');
199 										
200 										break;
201 									case ORYX.CONFIG.TYPE_COLOR:
202 										if(property.fill()) {
203 											if (svgElem.tagName.toLowerCase() === "stop"){
204 												if (prop){
205 													
206 													if (property.lightness() &&  property.lightness() !== 1){
207 														prop = ORYX.Utils.adjustLightness(prop, property.lightness());
208 													}
209 													
210 													svgElem.setAttributeNS(null, "stop-color", prop);
211 												
212 													// Adjust stop color of the others
213 													if (svgElem.parentNode.tagName.toLowerCase() === "radialgradient"){
214 														ORYX.Utils.adjustGradient(svgElem.parentNode, svgElem);
215 													}
216 												}
217 												
218 												// If there is no value, set opaque
219 												if (svgElem.parentNode.tagName.toLowerCase() === "radialgradient"){
220 													$A(svgElem.parentNode.getElementsByTagName('stop')).each(function(stop){
221 														stop.setAttributeNS(null, "stop-opacity", prop ? stop.getAttributeNS(ORYX.CONFIG.NAMESPACE_ORYX, 'default-stop-opacity') || 1 : 0);
222 													}.bind(this))
223 												}
224 											} else {
225 												svgElem.setAttributeNS(null, 'fill', prop);
226 											}
227 										}
228 										if(property.stroke()) {
229 											svgElem.setAttributeNS(null, 'stroke', prop);
230 										}
231 										break;
232 									case ORYX.CONFIG.TYPE_STRING:
233 										var label = this._labels[refId];
234 										if (label) {
235 											label.text(prop);
236 										}
237 										break;
238 									case ORYX.CONFIG.TYPE_INTEGER:
239 										var label = this._labels[refId];
240 										if (label) {
241 											label.text(prop);
242 										}
243 										break;
244 									case ORYX.CONFIG.TYPE_FLOAT:
245 										if(property.fillOpacity()) {
246 											svgElem.setAttributeNS(null, 'fill-opacity', prop);
247 										} 
248 										if(property.strokeOpacity()) {
249 											svgElem.setAttributeNS(null, 'stroke-opacity', prop);
250 										}
251 										if(!property.fillOpacity() && !property.strokeOpacity()) {
252 											var label = this._labels[refId];
253 											if (label) {
254 												label.text(prop);
255 											}
256 										}
257 										break;
258 									case ORYX.CONFIG.TYPE_URL:
259 									case ORYX.CONFIG.TYPE_DIAGRAM_LINK:
260 										//TODO what is the dafault path?
261 										var hrefAttr = svgElem.getAttributeNodeNS('http://www.w3.org/1999/xlink', 'xlink:href');
262 										if(hrefAttr) {
263 											hrefAttr.textContent = prop;
264 										} else {
265 											svgElem.setAttributeNS('http://www.w3.org/1999/xlink', 'xlink:href', prop);
266 										}	
267 										break;
268 								}
269 							}
270 						}).bind(this));
271 						
272 						
273 					}
274 					
275 				}
276 			}).bind(this));
277 			
278 			//update labels
279 			this._labels.values().each(function(label) {
280 				label.update();
281 			});
282 		}

479 	/**
480 	 * Overrides the UIObject.add method. Adds uiObject to the correct sub node.
481 	 * @param {UIObject} uiObject
482 	 * @param {Number} index
483 	 */
484 	add: function(uiObject, index, silent) {
485 		//parameter has to be an UIObject, but
486 		// must not be an Edge.
487 		if(uiObject instanceof ORYX.Core.UIObject 
488 			&& !(uiObject instanceof ORYX.Core.Edge)) {
489 			
490 			if (!(this.children.member(uiObject))) {
491 				//if uiObject is child of another parent, remove it from that parent.
492 				if(uiObject.parent) {
493 					uiObject.parent.remove(uiObject, true);
494 				}
495 
496 				//add uiObject to this Shape
497 				if(index != undefined)
498 					this.children.splice(index, 0, uiObject);
499 				else
500 					this.children.push(uiObject);
501 
502 				//set parent reference
503 				uiObject.parent = this;
504 
505 				//add uiObject.node to this.node depending on the type of uiObject
506 				var parent;
507 				if(uiObject instanceof ORYX.Core.Node) {
508 					parent = this.node.childNodes[0].childNodes[1];
509 					this.nodes.push(uiObject);
510 				} else if(uiObject instanceof ORYX.Core.Controls.Control) {
511 					var ctrls = this.node.childNodes[1];
512 					if(uiObject instanceof ORYX.Core.Controls.Docker) {
513 						parent = ctrls.childNodes[0];
514 						if (this.dockers.length >= 2){
515 							this.dockers.splice(index!==undefined?Math.min(index, this.dockers.length-1):this.dockers.length-1, 0, uiObject);
516 						} else {
517 							this.dockers.push(uiObject);
518 						}
519 					} else if(uiObject instanceof ORYX.Core.Controls.Magnet) {
520 						parent = ctrls.childNodes[1];
521 						this.magnets.push(uiObject);
522 					} else {
523 						parent = ctrls;
524 					}
525 				} else {	//UIObject
526 					parent = this.node;
527 				}
528 
529 				if(index != undefined && index < parent.childNodes.length)
530 					uiObject.node = parent.insertBefore(uiObject.node, parent.childNodes[index]);
531 				else
532 					uiObject.node = parent.appendChild(uiObject.node);
533 					
534 				this._changed();
535 				//uiObject.bounds.registerCallback(this._changedCallback);
536 				
537 				
538 				if(this.eventHandlerCallback && silent !== true)
539 					this.eventHandlerCallback({type:ORYX.CONFIG.EVENT_SHAPEADDED,shape:uiObject})
540 					
541 			} else {
542 
543 				ORYX.Log.warn("add: ORYX.Core.UIObject is already a child of this object.");
544 			}
545 		} else {
546 
547 			ORYX.Log.warn("add: Parameter is not of type ORYX.Core.UIObject.");
548 		}
549 	},

551 	/**
552 	 * Overrides the UIObject.remove method. Removes uiObject.
553 	 * @param {UIObject} uiObject
554 	 */
555 	remove: function(uiObject, silent) {
556 		//if uiObject is a child of this object, remove it.
557 		if (this.children.member(uiObject)) {
558 			//remove uiObject from children
559 			var parent = uiObject.parent;
560 
561 			this.children = this.children.without(uiObject);
562 
563 			//delete parent reference of uiObject
564 			uiObject.parent = undefined;
565 
566 			//delete uiObject.node from this.node
567 			if(uiObject instanceof ORYX.Core.Shape) {
568 				if(uiObject instanceof ORYX.Core.Edge) {
569 					uiObject.removeMarkers();
570 					uiObject.node = this.node.childNodes[0].childNodes[2].removeChild(uiObject.node);
571 				} else {
572 					uiObject.node = this.node.childNodes[0].childNodes[1].removeChild(uiObject.node);
573 					this.nodes = this.nodes.without(uiObject);
574 				}
575 			} else if(uiObject instanceof ORYX.Core.Controls.Control) {
576 				if (uiObject instanceof ORYX.Core.Controls.Docker) {
577 					uiObject.node = this.node.childNodes[1].childNodes[0].removeChild(uiObject.node);
578 					this.dockers = this.dockers.without(uiObject);
579 				} else if (uiObject instanceof ORYX.Core.Controls.Magnet) {
580 					uiObject.node = this.node.childNodes[1].childNodes[1].removeChild(uiObject.node);
581 					this.magnets = this.magnets.without(uiObject);
582 				} else {
583 					uiObject.node = this.node.childNodes[1].removeChild(uiObject.node);
584 				}
585 			}
586 
587 			if(this.eventHandlerCallback && silent !== true)
588 				this.eventHandlerCallback({type: ORYX.CONFIG.EVENT_SHAPEREMOVED, shape: uiObject, parent: parent});
589 			
590 			this._changed();
591 			//uiObject.bounds.unregisterCallback(this._changedCallback);
592 		} else {
593 
594 			ORYX.Log.warn("remove: ORYX.Core.UIObject is not a child of this object.");
595 		}

598 	/**
599 	 * Calculate the Border Intersection Point between two points
600 	 * @param {PointA}
601 	 * @param {PointB}
602 	 */
603 	getIntersectionPoint: function() {
604 			
605 		var pointAX, pointAY, pointBX, pointBY;
606 		
607 		// Get the the two Points	
608 		switch(arguments.length) {
609 			case 2:
610 				pointAX = arguments[0].x;
611 				pointAY = arguments[0].y;
612 				pointBX = arguments[1].x;
613 				pointBY = arguments[1].y;
614 				break;
615 			case 4:
616 				pointAX = arguments[0];
617 				pointAY = arguments[1];
618 				pointBX = arguments[2];
619 				pointBY = arguments[3];
620 				break;
621 			default:
622 				throw "getIntersectionPoints needs two or four arguments";
623 		}
624 		
625 		
626 		
627 		// Defined an include and exclude point
628 		var includePointX, includePointY, excludePointX, excludePointY;
629 
630 		var bounds = this.absoluteBounds();
631 		
632 		if(this.isPointIncluded(pointAX, pointAY, bounds)){
633 			includePointX = pointAX;
634 			includePointY = pointAY;
635 		} else {
636 			excludePointX = pointAX;
637 			excludePointY = pointAY;
638 		}
639 
640 		if(this.isPointIncluded(pointBX, pointBY, bounds)){
641 			includePointX = pointBX;
642 			includePointY = pointBY;
643 		} else {
644 			excludePointX = pointBX;
645 			excludePointY = pointBY;
646 		}
647 				
648 		// If there is no inclue or exclude Shape, than return
649 		if(!includePointX || !includePointY || !excludePointX || !excludePointY) {
650 			return undefined;
651 		}
652 
653 		var midPointX = 0;
654 		var midPointY = 0;		
655 		
656 		var refPointX, refPointY;

660 		//var ul = bounds.upperLeft();
661 		//var lr = bounds.lowerRight();
662 		
663 		var i = 0;
664 		
665 		while(true) {
666 			// Calculate the midpoint of the current to points	
667 			var midPointX = Math.min(includePointX, excludePointX) + ((Math.max(includePointX, excludePointX) - Math.min(includePointX, excludePointX)) / 2.0);
668 			var midPointY = Math.min(includePointY, excludePointY) + ((Math.max(includePointY, excludePointY) - Math.min(includePointY, excludePointY)) / 2.0);
669 			
670 			
671 			// Set the new midpoint by the means of the include of the bounds
672 			if(this.isPointIncluded(midPointX, midPointY, bounds)){
673 				includePointX = midPointX;
674 				includePointY = midPointY;
675 			} else {
676 				excludePointX = midPointX;
677 				excludePointY = midPointY;
678 			}			
679 			
680 			// Calc the length of the line
681 			var length = Math.sqrt(Math.pow(includePointX - excludePointX, 2) + Math.pow(includePointY - excludePointY, 2))
682 			// Calc a point one step from the include point
683 			refPointX = includePointX + ((excludePointX - includePointX) / length),
684 			refPointY = includePointY + ((excludePointY - includePointY) / length)
685 					
686 			
687 			// If the reference point not in the bounds, break
688 			if(!this.isPointIncluded(refPointX, refPointY, bounds)) {
689 				break
690 			}
691 							
692 			
693 		}
694 
695 		// Return the last includepoint
696 		return {x:refPointX , y:refPointY};
697 	},

844 		if(element instanceof Element) {
845 			var eid = element.getAttributeNS(null, 'id');
846 			if(eid && eid !== "") {
847 				element.setAttributeNS(null, 'id', this.id + eid);
848 			} else {
849 				element.setAttributeNS(null, 'id', this.id + "_" + this.id + "_" + idIndex);
850 				idIndex++;
851 			}
852 			
853 			// Replace URL in fill attribute
854 			var fill = element.getAttributeNS(null, 'fill');
855 			if (fill&&fill.include("url(#")){
856 				fill = fill.replace(/url\(#/g, 'url(#'+this.id);
857 				element.setAttributeNS(null, 'fill', fill);
858 			}
859 			
860 			if(element.hasChildNodes()) {
861 				for(var i = 0; i < element.childNodes.length; i++) {
862 					idIndex = this._adjustIds(element.childNodes[i], idIndex);
863 				}
864 			}
865 		}
866 		return idIndex;
