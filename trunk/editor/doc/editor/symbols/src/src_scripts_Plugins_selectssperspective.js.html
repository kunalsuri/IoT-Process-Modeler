 29 ORYX.Plugins.SelectStencilSetPerspective = {
 30 
 31 	facade: undefined,
 32 	
 33 	extensions : undefined,
 34 	
 35 	perspectives: undefined,
 36 
 37 	construct: function(facade) {
 38 		this.facade = facade;
 39 
 40 		var panel = new Ext.Panel({
 41 			cls:'selectssperspective',
 42 			border: false,
 43 			autoWidth:true,
 44 			autoScroll:true
 45 		});
 46 
 47 		var region = this.facade.addToRegion("west", panel);
 48 		
 49 		
 50 		var jsonObject = this.facade.getStencilSetExtensionDefinition();
 51 		
 52 		/* Determine available extensions */
 53 		this.extensions = {};
 54 		jsonObject.extensions.each(function(ext) {
 55 			this.extensions[ext.namespace] = ext;
 56 		}.bind(this));
 57 		
 58 		/* Determine available extensions */
 59 		this.perspectives = {};
 60 		jsonObject.perspectives.each(function(per) {
 61 			this.perspectives[per.namespace] = per;
 62 		}.bind(this));
 63 
 64 		this.facade.getStencilSets().values().each((function(sset) {
 65 
 66 			var validPerspectives = jsonObject.perspectives.findAll(function(perspective){
 67 				if(perspective.stencilset == sset.namespace()) return true;
 68 				else return false;
 69 			}); 
 70 			
 71 			
 72 			// If one perspective is defined, load this
 73 			if (validPerspectives.size() === 1) {
 74 				this.loadPerspective(validPerspectives.first().namespace);
 75 			
 76 			// If more than one perspective is defined, add a combobox and load the first one
 77 			} else if (validPerspectives.size() > 1) {
 78 				this.createPerspectivesCombobox(panel, sset, validPerspectives);
 79 			}
 80 
 81 		}).bind(this));
 82 					 
 83 
 84 	},
 85 
 86 	createPerspectivesCombobox: function(panel, stencilset, perspectives) {
 87 	
 88 		var lang = ORYX.I18N.Language.split("_").first();
 89 	
 90 		var data = [];
 91 		perspectives.each(function(perspective) {
 92 			data.push([perspective.namespace, (perspective["title_"+lang]||perspective.title).unescapeHTML(), perspective["description_"+lang]||perspective.description]);
 93 		});
 94 		
 95 	
 96 		var store = new Ext.data.SimpleStore({
 97 			fields: ['namespace', 'title', 'tooltip'],
 98 			data: data
 99 		});
100 	
101 		var combobox = new Ext.form.ComboBox({
102 			store			: store,
103 			displayField	: 'title',
104 			valueField		: 'namespace',
105 			forceSelection	: true,
106 			typeAhead		: true,
107 			mode			: 'local',
108 			allowBlank		: false,
109 			autoWidth		: true,
110 			triggerAction	: 'all',
111 			emptyText		: 'Select a perspective...',
112 			selectOnFocus	: true,
113 			tpl				: '<tpl for="."><div ext:qtip="{tooltip}" class="x-combo-list-item">{[(values.title||"").escapeHTML()]}</div></tpl>'
114 		});
115 		
116 		//panel.on("resize", function(){combobox.setWidth(panel.body.getWidth())});
117 		
118 		panel.add(combobox);
119 		panel.doLayout();
120 		
121 		combobox.on('beforeselect', this.onSelect ,this)
122 		
123 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_LOADED, function(){
124 					this.facade.getStencilSets().values().each(function(stencilset) {
125 						var ext = stencilset.extensions().values()
126 						if (ext.length > 0){
127 							var persp = perspectives.find(function(perspective){  
128 											return	(perspective.extensions && perspective.extensions.include(ext[0].namespace)) ||  			// Check if there is the extension part of the extension in the perspectives
129 													(perspective.addExtensions && perspective.addExtensions.any(function(add){ return 		// OR Check if the namespace if part of the addExtension part
130 														(add.ifIsLoaded === stencilset.namespace() && add.add == ext[0].namespace) || 
131 														(add.ifIsLoaded !== stencilset.namespace() && add["default"] === ext[0].namespace) 	// OR is some in the default
132 													})
133 												)
134 											})
135 							
136 							if (!persp){
137 								persp = perspectives.find(function(r){ return !(r.extensions instanceof Array) || r.extensions.length <= 0 })
138 							}
139 							
140 							if (persp) {
141 								combobox.setValue(data[perspectives.indexOf(persp)][1]);
142 								throw $break;
143 							}
144 						}
145 						// Force to load extension
146 						combobox.setValue(data[0][1]);
147 						this.loadPerspective(data[0][0]);
148 					}.bind(this));
149 				}.bind(this))
150 		
151 	},
152 	
153 	onSelect: function(combobox, record) {
154 		if (combobox.getValue() === record.get("namespace") || combobox.getValue() === record.get("title")){
155 			return;
156 		}
157 		
158 		this.loadPerspective(record.json[0]);
159 			
160 	},
161 	
162 	loadPerspective: function(ns){
163 		// If there is no namespace
164 		if (!ns){
165 			// unload all extensions
166 			this._loadExtensions([], [], true);
167 			return;
168 		}
169 		
170 		/* Get loaded stencil set extensions */
171 		var stencilSets = this.facade.getStencilSets();
172 		var loadedExtensions = new Object();
173 		var perspective = this.perspectives[ns];
174 		
175 		stencilSets.values().each(function(ss) { 
176 	    	ss.extensions().values().each(function(extension) {
177 				if(this.extensions[extension.namespace])
178 					loadedExtensions[extension.namespace] = extension;
179 			}.bind(this));
180 		}.bind(this));
181 		
182 		
183 		/* Determine extensions that are required for this perspective */
184 		var addExtensions = new Array();
185 		if(perspective.addExtensions||perspective.extensions) {
186 			[]
187 			 .concat(this.perspectives[ns].addExtensions||[])
188 			 .concat(this.perspectives[ns].extensions||[])
189 			 .compact()
190 			 .each(function(ext){
191 				if(!ext.ifIsLoaded) {
192 					addExtensions.push(this.extensions[ext]);
193 					return;
194 				}
195 				
196 				if(loadedExtensions[ext.ifIsLoaded] && this.extensions[ext.add]) {
197 					addExtensions.push(this.extensions[ext.add]);
198 				} else {
199 					if(ext["default"] && this.extensions[ext["default"]]) {
200 						addExtensions.push(this.extensions[ext["default"]]);
201 					}
202 				}
203 			}.bind(this));
204 		}
205 		
206 		/* Determine extension that are not allowed in this perspective */
207 		
208 		/* Check if flag to remove all other extension is set */
209 		if(this.perspectives[ns].removeAllExtensions) {	
210 			window.setTimeout(function(){
211 				this._loadExtensions(addExtensions, undefined, true);
212 			}.bind(this), 10);
213 			return;		
214 		}
215 		
216 		/* Check on specific extensions */
217 		var removeExtensions = new Array();
218 		if(perspective.removeExtensions) {
219 			perspective.removeExtensions.each(function(ns){
220 				if (loadedExtensions[ns])
221 					removeExtensions.push(this.extensions[ns]);
222 			}.bind(this));
223 		}
224 		
225 		if (perspective.extensions && !perspective.addExtensions && !perspective.removeExtensions) {
226 			var combined = [].concat(addExtensions).concat(removeExtensions).compact();
227 			$H(loadedExtensions).each(function(extension){
228 				var key = extension.key;
229 				if (!extension.value.includeAlways&&!combined.any(function(r){ return r.namespace == key })) {
230 					removeExtensions.push(this.extensions[key]);
231 				}
232 			}.bind(this))
233 		}
234 		
235 		window.setTimeout(function(){
236 			this._loadExtensions(addExtensions, removeExtensions, false);
237 		}.bind(this), 10);
238 	},
239 	
240 	/*
241 	 * Load all stencil set extensions specified in param extensions (key map: String -> Object)
242 	 * Unload all other extensions (method copied from addssextension plugin)
243 	 */
244 	_loadExtensions: function(addExtensions, removeExtensions, removeAll) {
245 		var stencilsets = this.facade.getStencilSets();
246 		
247 		var atLeastOne = false;
248 		
249 		// unload unselected extensions
250 		stencilsets.values().each(function(stencilset) {
251 			var unselected = stencilset.extensions().values().select(function(ext) { return addExtensions[ext.namespace] == undefined }); 
252 			if(removeAll) {
253 				unselected.each(function(ext) {
254 					stencilset.removeExtension(ext.namespace);
255 					atLeastOne = true;
256 				});
257 			} else {
258 				unselected.each(function(ext) {
259 					var remove = removeExtensions.find(function(remExt) {
260 						return ext.namespace === remExt.namespace;
261 					});
262 					
263 					if(remove) {
264 						stencilset.removeExtension(ext.namespace);
265 						atLeastOne = true;
266 					}
267 				});
268 			}
269 		});
270 		
271 		// load selected extensions
272 		addExtensions.each(function(extension) {
273 			
274 			var stencilset = stencilsets[extension["extends"]];
275 			
276 			if(stencilset) {
277 				// Load absolute
278 				if ((extension.definition || "").startsWith("/")) {
279 					stencilset.addExtension(extension.definition);
280 				// Load relative
281 				} else {
282 					stencilset.addExtension(ORYX.CONFIG.SS_EXTENSIONS_FOLDER + extension.definition);
283 				}
284 				atLeastOne = true;
285 			}
286 		}.bind(this));
287 		
288 		if (atLeastOne) {
289 			stencilsets.values().each(function(stencilset) {
290 				this.facade.getRules().initializeRules(stencilset);
291 			}.bind(this));
292 			this.facade.raiseEvent({
293 				type: ORYX.CONFIG.EVENT_STENCIL_SET_LOADED
294 			});
295 			var selection = this.facade.getSelection();
296 			this.facade.setSelection();
297 			this.facade.setSelection(selection);
298 		}
299 	}
300 
301 }
