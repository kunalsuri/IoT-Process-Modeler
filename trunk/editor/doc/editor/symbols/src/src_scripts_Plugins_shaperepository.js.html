 29 ORYX.Plugins.ShapeRepository = {
 30 
 31 	facade: undefined,
 32 
 33 	construct: function(facade) {
 34 		this.facade = facade;
 35 		this._currentParent;
 36 		this._canContain = undefined;
 37 		this._canAttach  = undefined;
 38 
 39 		this.shapeList = new Ext.tree.TreeNode({
 40 			
 41 		})
 42 
 43 		var panel = new Ext.tree.TreePanel({
 44             cls:'shaperepository',
 45 			loader: new Ext.tree.TreeLoader(),
 46 			root: this.shapeList,
 47 			autoScroll:true,
 48 			rootVisible: false,
 49 			lines: false,
 50 			anchors: '0, -30'
 51 		})
 52 		var region = this.facade.addToRegion("west", panel, ORYX.I18N.ShapeRepository.title);
 53 	
 54 		
 55 		// Create a Drag-Zone for Drag'n'Drop
 56 		var DragZone = new Ext.dd.DragZone(this.shapeList.getUI().getEl(), {shadow: !Ext.isMac});
 57 		DragZone.afterDragDrop = this.drop.bind(this, DragZone);
 58 		DragZone.beforeDragOver = this.beforeDragOver.bind(this, DragZone);
 59 		DragZone.beforeDragEnter = function(){this._lastOverElement = false; return true}.bind(this);
 60 		
 61 		// Load all Stencilssets
 62 		this.setStencilSets();
 63 		
 64 		this.facade.registerOnEvent(ORYX.CONFIG.EVENT_STENCIL_SET_LOADED, this.setStencilSets.bind(this));
 65 
 66 	},
 67 	
 68 	
 69 	/**
 70 	 * Load all stencilsets in the shaperepository
 71 	 */
 72 	setStencilSets: function() {
 73 		// Remove all childs
 74 		var child = this.shapeList.firstChild;
 75 		while(child) {
 76 			this.shapeList.removeChild(child);
 77 			child = this.shapeList.firstChild;
 78 		}
 79 
 80 		// Go thru all Stencilsets and stencils
 81 		this.facade.getStencilSets().values().each((function(sset) {
 82 			
 83 			// For each Stencilset create and add a new Tree-Node
 84 			var stencilSetNode
 85 			
 86 			var typeTitle = sset.title();
 87 			var extensions = sset.extensions();
 88 			if (extensions && extensions.size() > 0) {
 89 				typeTitle += " / " + ORYX.Core.StencilSet.getTranslation(extensions.values()[0], "title");
 90 			} 
 91 			
 92 			this.shapeList.appendChild(stencilSetNode = new Ext.tree.TreeNode({
 93 				text:typeTitle, 			// Stencilset Name
 94 				allowDrag:false,
 95         		allowDrop:false,           
 96 				iconCls:'headerShapeRepImg',
 97 	            cls:'headerShapeRep',
 98 				singleClickExpand:true}));
 99 			
100 			stencilSetNode.render();
101 			stencilSetNode.expand();				
102 			// Get Stencils from Stencilset
103 			var stencils = sset.stencils(this.facade.getCanvas().getStencil(),
104 										 this.facade.getRules());	
105 			var treeGroups = new Hash();
106 			
107 			// Sort the stencils according to their position and add them to the repository
108 			stencils = stencils.sortBy(function(value) { return value.position(); } );
109 			stencils.each((function(value) {
110 				
111 				// Show stencils in no group if there is less than 10 shapes
112 				if(stencils.length <= ORYX.CONFIG.MAX_NUM_SHAPES_NO_GROUP) {
113 					this.createStencilTreeNode(stencilSetNode, value);	
114 					return;					
115 				}
116 				
117 				// Get the groups name
118 				var groups = value.groups();
119 				
120 				// For each Group-Entree
121 				groups.each((function(group) {
122 					
123 					// If there is a new group
124 					if(!treeGroups[group]) {
125 						// Create a new group
126 						treeGroups[group] = new Ext.tree.TreeNode({
127 							text:group,					// Group-Name
128 							allowDrag:false,
129         					allowDrop:false,            
130 							iconCls:'headerShapeRepImg', // Css-Class for Icon
131 				            cls:'headerShapeRepChild',  // CSS-Class for Stencil-Group
132 							singleClickExpand:true});
133 						
134 						// Add the Group to the ShapeRepository
135 						stencilSetNode.appendChild(treeGroups[group]);
136 						treeGroups[group].render();	
137 					}
138 					
139 					// Create the Stencil-Tree-Node
140 					this.createStencilTreeNode(treeGroups[group], value);	
141 					
142 				}).bind(this));
143 				
144 				
145 				// If there is no group
146 				if(groups.length == 0) {
147 					// Create the Stencil-Tree-Node
148 					this.createStencilTreeNode(stencilSetNode, value);						
149 				}
150 	
151 			}).bind(this));
152 		}).bind(this));
153 			
154 		if (this.shapeList.firstChild.firstChild) {
155 			this.shapeList.firstChild.firstChild.expand(false, true);
156 		}	
157 	},
158 
159 	createStencilTreeNode: function(parentTreeNode, stencil) {
160 
161 		// Create and add the Stencil to the Group
162 		var newElement = new Ext.tree.TreeNode({
163 				text:		stencil.title(), 		// Text of the stencil
164 				icon:		stencil.icon(),			// Icon of the stencil
165 				allowDrag:	false,					// Don't use the Drag and Drop of Ext-Tree
166 				allowDrop:	false,
167 				iconCls:	'ShapeRepEntreeImg', 	// CSS-Class for Icon
168 				cls:		'ShapeRepEntree'		// CSS-Class for the Tree-Entree
169 				});
170 
171 		parentTreeNode.appendChild(newElement);		
172 		newElement.render();	
173 				
174 		var ui = newElement.getUI();
175 		
176 		// Set the tooltip
177 		ui.elNode.setAttributeNS(null, "title", stencil.description());
178 		
179 		// Register the Stencil on Drag and Drop
180 		Ext.dd.Registry.register(ui.elNode, {
181 				node: 		ui.node,
182 		        handles: 	[ui.elNode, ui.textNode].concat($A(ui.elNode.childNodes)), // Set the Handles
183 		        isHandle: 	false,
184 				type:		stencil.id(),			// Set Type of stencil 
185 				namespace:	stencil.namespace()		// Set Namespace of stencil
186 				});
187 								
188 	},
189 	
190 	drop: function(dragZone, target, event) {
191 		
192 		this._lastOverElement = undefined;
193 		
194 		// Hide the highlighting
195 		this.facade.raiseEvent({type: ORYX.CONFIG.EVENT_HIGHLIGHT_HIDE, highlightId:'shapeRepo.added'});
196 		this.facade.raiseEvent({type: ORYX.CONFIG.EVENT_HIGHLIGHT_HIDE, highlightId:'shapeRepo.attached'});
197 		
198 		// Check if drop is allowed
199 		var proxy = dragZone.getProxy()
200 		if(proxy.dropStatus == proxy.dropNotAllowed) { return }
201 		
202 		// Check if there is a current Parent
203 		if(!this._currentParent) { return }
204 		
205 		var option = Ext.dd.Registry.getHandle(target.DDM.currentTarget);
206 		
207 		var xy = event.getXY();
208 		var pos = {x: xy[0], y: xy[1]};
209 
210 		var a = this.facade.getCanvas().node.getScreenCTM();
211 
212 		// Correcting the UpperLeft-Offset
213 		pos.x -= a.e; pos.y -= a.f;
214 		// Correcting the Zoom-Faktor
215 		pos.x /= a.a; pos.y /= a.d;
216 		// Correting the ScrollOffset
217 		pos.x -= document.documentElement.scrollLeft;
218 		pos.y -= document.documentElement.scrollTop;
219 		// Correct position of parent
220 		var parentAbs = this._currentParent.absoluteXY();
221 		pos.x -= parentAbs.x;
222 		pos.y -= parentAbs.y;
223 
224 		// Set position
225 		option['position'] = pos
226 		
227 		// Set parent
228 		if( this._canAttach &&  this._currentParent instanceof ORYX.Core.Node ){
229 			option['parent'] = undefined;	
230 		} else {
231 			option['parent'] = this._currentParent;
232 		}
233 		
234 		
235 		var commandClass = ORYX.Core.Command.extend({
236 			construct: function(option, currentParent, canAttach, position, facade){
237 				this.option = option;
238 				this.currentParent = currentParent;
239 				this.canAttach = canAttach;
240 				this.position = position;
241 				this.facade = facade;
242 				this.selection = this.facade.getSelection();
243 				this.shape;
244 				this.parent;
245 			},			
246 			execute: function(){
247 				if (!this.shape) {
248 					this.shape 	= this.facade.createShape(option);
249 					this.parent = this.shape.parent;
250 				} else {
251 					this.parent.add(this.shape);
252 				}
253 					
254 				
255 				
256 				if( this.canAttach &&  this.currentParent instanceof ORYX.Core.Node && this.shape.dockers.length > 0){
257 					
258 					var docker = this.shape.dockers[0];
259 		
260 					if( this.currentParent.parent instanceof ORYX.Core.Node ) {
261 						this.currentParent.parent.add( docker.parent );
262 					}
263 												
264 					docker.bounds.centerMoveTo( this.position );
265 					docker.setDockedShape( this.currentParent );
266 					//docker.update();	
267 				}
268 		
269 				//this.currentParent.update();
270 				//this.shape.update();
271 
272 				this.facade.setSelection([this.shape]);
273 				this.facade.getCanvas().update();
274 				this.facade.updateSelection();
275 				
276 			},
277 			rollback: function(){
278 				this.facade.deleteShape(this.shape);
279 				
280 				//this.currentParent.update();
281 
282 				this.facade.setSelection(this.selection.without(this.shape));
283 				this.facade.getCanvas().update();
284 				this.facade.updateSelection();
285 				
286 			}
287 		});
288 							
289 		var position = this.facade.eventCoordinates( event.browserEvent );	
290 					
291 		var command = new commandClass(option, this._currentParent, this._canAttach, position, this.facade);
292 		
293 		this.facade.executeCommands([command]);
294 		
295 		this._currentParent = undefined;
296 	},
297 
298 	beforeDragOver: function(dragZone, target, event){
299 
300 		var coord = this.facade.eventCoordinates(event.browserEvent);
301 		var aShapes = this.facade.getCanvas().getAbstractShapesAtPosition( coord );
302 
303 		if(aShapes.length <= 0) {
304 			
305 				var pr = dragZone.getProxy();
306 				pr.setStatus(pr.dropNotAllowed);
307 				pr.sync();
308 				
309 				return false;
310 		}	
311 		
312 		var el = aShapes.last();
313 	
314 		
315 		if(aShapes.lenght == 1 && aShapes[0] instanceof ORYX.Core.Canvas) {
316 			
317 			return false;
318 			
319 		} else {
320 			// check containment rules
321 			var option = Ext.dd.Registry.getHandle(target.DDM.currentTarget);
322 
323 			var stencilSet = this.facade.getStencilSets()[option.namespace];
324 
325 			var stencil = stencilSet.stencil(option.type);
326 
327 			if(stencil.type() === "node") {
328 				
329 				var parentCandidate = aShapes.reverse().find(function(candidate) {
330 					return (candidate instanceof ORYX.Core.Canvas 
331 							|| candidate instanceof ORYX.Core.Node
332 							|| candidate instanceof ORYX.Core.Edge);
333 				});
334 				
335 				if(  parentCandidate !== this._lastOverElement){
336 					
337 					this._canAttach  = undefined;
338 					this._canContain = undefined;
339 					
340 				}
341 				
342 				if( parentCandidate ) {
343 					//check containment rule					
344 						
345 					if (!(parentCandidate instanceof ORYX.Core.Canvas) && parentCandidate.isPointOverOffset(coord.x, coord.y) && this._canAttach == undefined) {
346 					
347 						this._canAttach = this.facade.getRules().canConnect({
348 												sourceShape: parentCandidate,
349 												edgeStencil: stencil,
350 												targetStencil: stencil
351 											});
352 						
353 						if( this._canAttach ){
354 							// Show Highlight
355 							this.facade.raiseEvent({
356 								type: ORYX.CONFIG.EVENT_HIGHLIGHT_SHOW,
357 								highlightId: "shapeRepo.attached",
358 								elements: [parentCandidate],
359 								style: ORYX.CONFIG.SELECTION_HIGHLIGHT_STYLE_RECTANGLE,
360 								color: ORYX.CONFIG.SELECTION_VALID_COLOR
361 							});
362 							
363 							this.facade.raiseEvent({
364 								type: ORYX.CONFIG.EVENT_HIGHLIGHT_HIDE,
365 								highlightId: "shapeRepo.added"
366 							});
367 							
368 							this._canContain	= undefined;
369 						} 					
370 						
371 					}
372 					
373 					if(!(parentCandidate instanceof ORYX.Core.Canvas) && !parentCandidate.isPointOverOffset(coord.x, coord.y)){
374 						this._canAttach 	= this._canAttach == false ? this._canAttach : undefined;						
375 					}
376 					
377 					if( this._canContain == undefined && !this._canAttach) {
378 											
379 						this._canContain = this.facade.getRules().canContain({
380 															containingShape:parentCandidate, 
381 															containedStencil:stencil
382 															});
383 															
384 						// Show Highlight
385 						this.facade.raiseEvent({
386 															type:		ORYX.CONFIG.EVENT_HIGHLIGHT_SHOW, 
387 															highlightId:'shapeRepo.added',
388 															elements:	[parentCandidate],
389 															color:		this._canContain ? ORYX.CONFIG.SELECTION_VALID_COLOR : ORYX.CONFIG.SELECTION_INVALID_COLOR
390 														});	
391 						this.facade.raiseEvent({
392 															type: 		ORYX.CONFIG.EVENT_HIGHLIGHT_HIDE,
393 															highlightId:"shapeRepo.attached"
394 														});						
395 					}
396 						
397 				
398 					
399 					this._currentParent = this._canContain || this._canAttach ? parentCandidate : undefined;
400 					this._lastOverElement = parentCandidate;
401 					var pr = dragZone.getProxy();
402 					pr.setStatus(this._currentParent ? pr.dropAllowed : pr.dropNotAllowed );
403 					pr.sync();
404 	
405 				} 
406 			} else { //Edge
407 				this._currentParent = this.facade.getCanvas();
408 				var pr = dragZone.getProxy();
409 				pr.setStatus(pr.dropAllowed);
410 				pr.sync();
411 			}		
412 		}
413 		
414 		
415 		return false
416 	}	
417 }
