 24 function printf() {
 25 	
 26 	var result = arguments[0];
 27 	for (var i=1; i<arguments.length; i++)
 28 		result = result.replace('%' + (i-1), arguments[i]);
 29 	return result;
 30 }

242 	_loadPlugins: function() {
243 
244 		// load plugin configuration file.
245 		var source = ORYX.CONFIG.PLUGINS_CONFIG;
246 
247 		ORYX.Log.debug("Loading plugin configuration from '%0'.", source);
248 	
249 		new Ajax.Request(source, {
250 			asynchronous: false,
251 			method: 'get',
252 			onSuccess: function(result) {
253 
254 				/*
255 				 * This is the method that is being called when the plugin
256 				 * configuration was successfully loaded from the server. The
257 				 * file has to be processed and the contents need to be
258 				 * considered for further plugin requireation.
259 				 */
260 				
261 				ORYX.Log.info("Plugin configuration file loaded.");
262 		
263 				// get plugins.xml content
264 				var resultXml = result.responseXML;
265 				
266 				// TODO: Describe how properties are handled.
267 				// Get the globale Properties
268 				var globalProperties = [];
269 				var preferences = $A(resultXml.getElementsByTagName("properties"));
270 				preferences.each( function(p) {
271 
272 					var props = $A(p.childNodes);
273 					props.each( function(prop) {
274 						var property = new Hash(); 
275 						
276 						// get all attributes from the node and set to global properties
277 						var attributes = $A(prop.attributes)
278 						attributes.each(function(attr){property[attr.nodeName] = attr.nodeValue});				
279 						if(attributes.length > 0) { globalProperties.push(property) };				
280 					});
281 				});
282 
283 				
284 				// TODO Why are we using XML if we don't respect structure anyway?
285 				// for each plugin element in the configuration..
286 				var plugin = resultXml.getElementsByTagName("plugin");
287 				$A(plugin).each( function(node) {
288 					
289 					// get all element's attributes.
290 					// TODO: What about: var pluginData = $H(node.attributes) !?
291 					var pluginData = new Hash();
292 					$A(node.attributes).each( function(attr){
293 						pluginData[attr.nodeName] = attr.nodeValue});				
294 					
295 					// ensure there's a name attribute.
296 					if(!pluginData['name']) {
297 						ORYX.Log.error("A plugin is not providing a name. Ingnoring this plugin.");
298 						return;
299 					}
300 
301 					// ensure there's a source attribute.
302 					if(!pluginData['source']) {
303 						ORYX.Log.error("Plugin with name '%0' doesn't provide a source attribute.", pluginData['name']);
304 						return;
305 					}
306 					
307 					// Get all private Properties
308 					var propertyNodes = node.getElementsByTagName("property");
309 					var properties = [];
310 					$A(propertyNodes).each(function(prop) {
311 						var property = new Hash(); 
312 						
313 						// Get all Attributes from the Node			
314 						var attributes = $A(prop.attributes)
315 						attributes.each(function(attr){property[attr.nodeName] = attr.nodeValue});				
316 						if(attributes.length > 0) { properties.push(property) };	
317 					
318 					});
319 					
320 					// Set all Global-Properties to the Properties
321 					properties = properties.concat(globalProperties);
322 					
323 					// Set Properties to Plugin-Data
324 					pluginData['properties'] = properties;
325 					
326 					// Get the RequieredNodes
327 					var requireNodes = node.getElementsByTagName("requires");
328 					var requires;
329 					$A(requireNodes).each(function(req) {			
330 						var namespace = $A(req.attributes).find(function(attr){ return attr.name == "namespace"})
331 						if( namespace && namespace.nodeValue ){
332 							if( !requires ){
333 								requires = {namespaces:[]}
334 							}
335 						
336 							requires.namespaces.push(namespace.nodeValue)
337 						} 
338 					});					
339 					
340 					// Set Requires to the Plugin-Data, if there is one
341 					if( requires ){
342 						pluginData['requires'] = requires;
343 					}
344 
345 
346 					// Get the RequieredNodes
347 					var notUsesInNodes = node.getElementsByTagName("notUsesIn");
348 					var notUsesIn;
349 					$A(notUsesInNodes).each(function(not) {			
350 						var namespace = $A(not.attributes).find(function(attr){ return attr.name == "namespace"})
351 						if( namespace && namespace.nodeValue ){
352 							if( !notUsesIn ){
353 								notUsesIn = {namespaces:[]}
354 							}
355 						
356 							notUsesIn.namespaces.push(namespace.nodeValue)
357 						} 
358 					});					
359 					
360 					// Set Requires to the Plugin-Data, if there is one
361 					if( notUsesIn ){
362 						pluginData['notUsesIn'] = notUsesIn;
363 					}		
364 					
365 								
366 					var url = ORYX.PATH + ORYX.CONFIG.PLUGINS_FOLDER + pluginData['source'];
367 		
368 					ORYX.Log.debug("Requireing '%0'", url);
369 		
370 					// Add the Script-Tag to the Site
371 					//Kickstart.require(url);
372 		
373 					ORYX.Log.info("Plugin '%0' successfully loaded.", pluginData['name']);
374 		
375 					// Add the Plugin-Data to all available Plugins
376 					ORYX.availablePlugins.push(pluginData);
377 		
378 				});
379 		
380 			},
381 			onFailure:this._loadPluginsOnFails
382 		});
383 
384 	},
385
